### 8.3. Topology Visualization

**Файлы:**
```
web/kubernetes/components/topology-map.js
```

**Функционал:**
- [ ] Визуализация связей между ресурсами
- [ ] Namespace → Deployment → Pod → Service
- [ ] Interactive graph (zoom, pan)
- [ ] Status indicators на узлах
- [ ] Click для деталей

**Пример кода:**
```javascript
class TopologyMap {
  constructor(containerId, options = {}) {
    this.container = document.getElementById(containerId);
    this.namespace = options.namespace;
    this.graph = null;

    this.init();
  }

  async init() {
    // Загружаем данные
    const [deployments, pods, services] = await Promise.all([
      k8s.listDeployments(this.namespace),
      k8s.listPods(this.namespace),
      k8s.listServices(this.namespace)
    ]);

    // Строим graph
    const nodes = [];
    const edges = [];

    // Добавляем Deployment узлы
    deployments.forEach(dep => {
      nodes.push({
        id: `deployment/${dep.name}`,
        type: 'deployment',
        label: dep.name,
        status: this.getDeploymentStatus(dep),
        replicas: `${dep.replicas_ready}/${dep.replicas}`
      });
    });

    // Добавляем Pod узлы
    pods.forEach(pod => {
      nodes.push({
        id: `pod/${pod.name}`,
        type: 'pod',
        label: pod.name,
        status: pod.phase
      });
    });

    // Добавляем Service узлы
    services.forEach(svc => {
      nodes.push({
        id: `service/${svc.name}`,
        type: 'service',
        label: svc.name,
        status: 'active'
      });
    });

    // Добавляем связи
    deployments.forEach(dep => {
      // Deployment → Pods (по label selector)
      const depPods = pods.filter(pod => this.matchSelector(pod, dep.selector));
      depPods.forEach(pod => {
        edges.push({
          source: `deployment/${dep.name}`,
          target: `pod/${pod.name}`,
          type: 'manages'
        });
      });

      // Service → Deployment (по selector)
      const depService = services.find(svc => this.matchSelector(dep, svc.selector));
      if (depService) {
        edges.push({
          source: `service/${depService.name}`,
          target: `deployment/${dep.name}`,
          type: 'routes'
        });
      }
    });

    // Рисуем graph с Cytoscape.js
    this.render(nodes, edges);
  }

  render(nodes, edges) {
    this.graph = cytoscape({
      container: this.container,
      elements: [
        ...nodes.map(n => ({ data: n })),
        ...edges.map(e => ({ data: e }))
      ],
      style: [
        {
          selector: 'node[type="deployment"]',
          style: {
            'background-color': '#3498db',
            'label': 'data(label)'
          }
        },
        {
          selector: 'node[type="pod"]',
          style: {
            'background-color': '#2ecc71',
            'label': 'data(label)'
          }
        },
        {
          selector: 'node[type="service"]',
          style: {
            'background-color': '#e74c3c',
            'label': 'data(label)'
          }
        },
        {
          selector: 'edge',
          style: {
            'width': 2,
            'line-color': '#95a5a6',
            'target-arrow-color': '#95a5a6',
            'target-arrow-shape': 'triangle'
          }
        }
      ],
      layout: {
        name: 'dagre',
        rankDir: 'TB',
        animate: true
      }
    });
  }
}
```
