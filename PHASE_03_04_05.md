/// Список secrets
/// GET /api/kubernetes/secrets
pub async fn list_secrets(
    State(state): State<Arc<AppState>>,
    Query(query): Query<ListQuery>,
) -> Result<Json<Vec<serde_json::Value>>> {
    let client = state.kubernetes_client()?;
    let api: Api<Secret> = if let Some(ns) = query.namespace {
        client.api::<Secret>(Some(&ns))
    } else {
        client.api_all()
    };

    let secrets = api
        .list(&ListParams::default())
        .await
        .map_err(|e| Error::KubernetesError(e.to_string()))?;

    let result = secrets
        .items
        .iter()
        .map(|secret| {
            let meta = &secret.metadata;
            let secret_type = &secret.type_;

            // Получаем ключи (но не значения!)
            let keys = secret.data
                .as_ref()
                .map(|data| data.keys().cloned().collect::<Vec<_>>())
                .unwrap_or_default();

            serde_json::json!({
                "name": meta.name.clone().unwrap_or_else(|| "unknown".to_string()),
                "namespace": meta.namespace.clone().unwrap_or_else(|| "default".to_string()),
                "type": secret_type.as_deref().unwrap_or("Opaque"),
                "keys": keys,
                "created_at": meta.creation_timestamp.as_ref().map(|t| t.to_rfc3339()).unwrap_or_else(|| "unknown".to_string()),
                "labels": meta.labels.clone().unwrap_or_default(),
            })
        })
        .collect();

    Ok(Json(result))
}

/// Детали secret с опциональным декодированием
/// GET /api/kubernetes/namespaces/{ns}/secrets/{name}
pub async fn get_secret(
    State(state): State<Arc<AppState>>,
    Path((namespace, name)): Path<(String, String)>,
    Query(query): Query<SecretQuery>,
) -> Result<Json<serde_json::Value>> {
    let client = state.kubernetes_client()?;
    let api: Api<Secret> = client.api::<Secret>(Some(&namespace));

    let secret = api
        .get(&name)
        .await
        .map_err(|e| Error::KubernetesError(e.to_string()))?;

    let show_values = query.decode.unwrap_or(false);

    // Декодируем значения только если запрошено
    let data = if show_values {
        secret.data.as_ref().map(|data| {
            data.iter().map(|(k, v)| {
                let decoded = String::from_utf8_lossy(&v.0).to_string();
                (k.clone(), decoded)
            }).collect::<HashMap<_, _>>()
        })
    } else {
        // Маскируем значения
        secret.data.as_ref().map(|data| {
            data.iter().map(|(k, v)| {
                (k.clone(), "***MASKED***".to_string())
            }).collect::<HashMap<_, _>>()
        })
    };

    Ok(Json(serde_json::json!({
        "name": name,
        "namespace": namespace,
        "type": secret.type_,
        "data": data,
        "labels": secret.metadata.labels,
        "annotations": secret.metadata.annotations,
        "created_at": secret.metadata.creation_timestamp,
    })))
}
```

class IngressVisualization {
  constructor(containerId, ingress) {
    this.container = document.getElementById(containerId);
    this.ingress = ingress;
    this.render();
  }

  render() {
    const rules = this.ingress.spec.rules || [];
    const tls = this.ingress.spec.tls || [];

    this.container.innerHTML = `
      <div class="ingress-diagram">
        <div class="ingress-node">
          <div class="ingress-label">Ingress: ${this.ingress.metadata.name}</div>
        </div>

        ${rules.map((rule, idx) => `
          <div class="ingress-rule">
            <div class="rule-host">
              <i class="fa-solid fa-globe"></i>
              ${rule.host || '*'}
            </div>

            ${(rule.http.paths || []).map(path => `
              <div class="rule-path">
                <div class="path-value">
                  <i class="fa-solid fa-route"></i>
                  ${path.path || '/'} → ${path.backend.service.name}:${path.backend.service.port.number}
                </div>
              </div>
            `).join('')}
          </div>
        `).join('')}

        ${tls.length > 0 ? `
          <div class="ingress-tls">
            <h4><i class="fa-solid fa-lock"></i> TLS</h4>
            ${tls.map(t => `
              <div class="tls-entry">
                <div>Hosts: ${(t.hosts || []).join(', ')}</div>
                <div>Secret: ${t.secretName}</div>
              </div>
            `).join('')}
          </div>
        ` : ''}
      </div>
    `;
  }
}
```
